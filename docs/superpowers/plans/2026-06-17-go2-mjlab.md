# Go2 mjlab Migration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Migrate Go2 locomotion training from CPU MuJoCo to GPU-accelerated mjlab, achieving 20-50x speedup with full Teacher → Student → ONNX pipeline.

**Architecture:** Extension project under `applications/go2_mjlab/` using mjlab's manager-based API. Go2 task registered via `register_mjlab_task()`. Rewards/observations/DR as plain functions returning `[B]` tensors. RSL-RL handles PPO training loop. Offline distillation as separate PyTorch script.

**Tech Stack:** mjlab >= 1.4.0, mujoco-warp >= 3.9.0, rsl-rl-lib >= 5.4.0, torch >= 2.7.0, warp-lang >= 1.14.0, tyro

**API Notes (from unitree_rl_mjlab reference):**
- Task discovery via pyproject.toml `[project.entry-points."mjlab.tasks"]`, not just `import_packages`
- Go2 actuator physics: Hip/Thigh stiffness=20, damping=1, effort=23.5; Calf stiffness=40, damping=2, effort=45
- Joint config uses regex patterns: `".*thigh_joint": 0.9`, `".*calf_joint": -1.8`
- Go2 gait phase offsets: `[0.0, 0.5, 0.5, 0.0]` (trot: FR+RL together, FL+RR together)
- Built-in gait phase observation: sin/cos clock signal (period=0.6s)
- Our training hyperparams (hidden_dims, lr, action_scale, reward_weights) kept from existing go2_locomotion

# src/tasks/velocity/rl_cfg.py
from mjlab.rl import RslRlOnPolicyRunnerCfg, RslRlModelCfg, RslRlPpoAlgorithmCfg


def make_go2_ppo_runner_cfg() -> RslRlOnPolicyRunnerCfg:
    """PPO config — our hyperparams from go2_locomotion.
    
    Differences from unitree default (512,256,128 / lr=1e-3 / adaptive):
    - Smaller network (128,128) — sufficient for Go2, trains faster
    - Lower LR with fixed schedule — matches our linear decay approach
    - Higher entropy coef (0.02) — more exploration early on
    - More steps per env (48 vs 24) — longer rollouts
    """
    return RslRlOnPolicyRunnerCfg(
        actor=RslRlModelCfg(
            hidden_dims=(128, 128),
            activation="elu",
            obs_normalization=True,
            distribution_cfg={
                "class_name": "GaussianDistribution",
                "init_std": 1.0,
                "std_type": "scalar",
            },
        ),
        critic=RslRlModelCfg(
            hidden_dims=(128, 128),
            activation="elu",
            obs_normalization=True,
        ),
        algorithm=RslRlPpoAlgorithmCfg(
            learning_rate=3e-4,
            schedule="fixed",
            gamma=0.99,
            lam=0.95,
            entropy_coef=0.02,
            clip_param=0.2,
            max_grad_norm=1.0,
            num_learning_epochs=5,
            num_mini_batches=4,
            value_loss_coef=1.0,
        ),
        experiment_name="go2_mjlab_teacher",
        save_interval=500,
        num_steps_per_env=48,
        max_iterations=5000,
    )

All reward functions take `env: ManagerBasedRlEnv` as first arg and return a `[B]` shaped tensor.

```python
# src/tasks/velocity/mdp/rewards.py
import torch
from mjlab.envs import ManagerBasedRlEnv
from mjlab.entity import SceneEntityCfg

_DEFAULT_ASSET_CFG = SceneEntityCfg("robot")


def track_lin_vel_exp(
    env: ManagerBasedRlEnv,
    command_name: str,
    std: float,
    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
) -> torch.Tensor:
    """Exponential reward for tracking commanded linear velocity."""
    asset = env.scene[asset_cfg.name]
    vel = asset.data.root_lin_vel_b[:, :2]  # [B, 2] local xy
    cmd = env.command_manager.get_command(command_name)[:, :2]
    error = torch.sum((vel - cmd) ** 2, dim=1)
    return torch.exp(-error / std**2)


def track_ang_vel_exp(
    env: ManagerBasedRlEnv,
    command_name: str,
    std: float,
    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
) -> torch.Tensor:
    """Exponential reward for tracking commanded angular velocity."""
    asset = env.scene[asset_cfg.name]
    ang_vel_z = asset.data.root_ang_vel_b[:, 2]
    cmd_yaw = env.command_manager.get_command(command_name)[:, 2]
    error = (ang_vel_z - cmd_yaw) ** 2
    return torch.exp(-error / std**2)


def forward_velocity(
    env: ManagerBasedRlEnv,
    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
) -> torch.Tensor:
    """Raw forward velocity in body frame (faster is always better)."""
    asset = env.scene[asset_cfg.name]
    return asset.data.root_lin_vel_b[:, 0]


def feet_air_time(
    env: ManagerBasedRlEnv,
    threshold: float,
    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
) -> torch.Tensor:
    """Reward feet spending time in the air (above threshold)."""
    asset = env.scene[asset_cfg.name]
    air_time = asset.data.feet_air_time  # [B, 4]
    contact = asset.data.feet_contact    # [B, 4] bool
    # Reward on landing: air_time - threshold, clipped to 0
    first_contact = contact & (air_time > 0)
    reward_per_foot = torch.clamp(air_time - threshold, min=0.0) * first_contact.float()
    return reward_per_foot.sum(dim=1)


class gait_schedule_reward:
    """Reward feet following a trot gait schedule (diagonal pairs in anti-phase)."""

    def __init__(self, cfg, env: ManagerBasedRlEnv):
        self.step_dt = env.step_dt
        self.time = torch.zeros(env.num_envs, device=env.device)

    def __call__(
        self,
        env: ManagerBasedRlEnv,
        trot_period: float,
        asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
    ) -> torch.Tensor:
        self.time += self.step_dt
        asset = env.scene[asset_cfg.name]
        contact = asset.data.feet_contact.float()  # [B, 4]: FL, FR, RL, RR
        phase = (self.time / trot_period) % 1.0  # [B]

        # Trot: FL+RR in phase, FR+RL in anti-phase
        # phase 0-0.5: FL+RR should be in swing (not contact)
        # phase 0.5-1: FR+RL should be in swing
        swing_mask = (phase < 0.5).float()  # [B]

        # Desired contact pattern
        # When swing_mask=1: FL=0, FR=1, RL=1, RR=0
        # When swing_mask=0: FL=1, FR=0, RL=0, RR=1
        desired = torch.stack([
            1 - swing_mask,  # FL
            swing_mask,      # FR
            swing_mask,      # RL
            1 - swing_mask,  # RR
        ], dim=1)

        match = 1.0 - torch.abs(contact - desired)
        return match.mean(dim=1)

    def reset(self, env_ids: torch.Tensor) -> None:
        self.time[env_ids] = 0.0


def gait_symmetry_reward(
    env: ManagerBasedRlEnv,
    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
) -> torch.Tensor:
    """Reward diagonal pair symmetry (FL+RR vs FR+RL in anti-phase)."""
    asset = env.scene[asset_cfg.name]
    contact = asset.data.feet_contact.float()  # [B, 4]
    # Diagonal pairs should be opposite
    pair1_match = torch.abs(contact[:, 0] - contact[:, 3])  # FL vs RR same
    pair2_match = torch.abs(contact[:, 1] - contact[:, 2])  # FR vs RL same
    cross = torch.abs(contact[:, 0] - contact[:, 1])         # FL vs FR opposite
    return (1.0 - pair1_match) + (1.0 - pair2_match) + cross


def all_feet_contact(
    env: ManagerBasedRlEnv,
    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
) -> torch.Tensor:
    """Penalty when all four feet are in contact (lurching/standing)."""
    asset = env.scene[asset_cfg.name]
    contact = asset.data.feet_contact  # [B, 4] bool
    all_down = contact.all(dim=1).float()
    return all_down


def base_height_reward(
    env: ManagerBasedRlEnv,
    target: float,
    sigma: float,
    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
) -> torch.Tensor:
    """Gaussian reward for maintaining target base height."""
    asset = env.scene[asset_cfg.name]
    height = asset.data.root_pos_w[:, 2]
    error = (height - target) ** 2
    return torch.exp(-error / sigma**2)


def flat_orientation(
    env: ManagerBasedRlEnv,
    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
) -> torch.Tensor:
    """Penalty for non-flat orientation (sum of squared roll+pitch gravity components)."""
    asset = env.scene[asset_cfg.name]
    gravity_b = asset.data.projected_gravity_b  # [B, 3]
    return gravity_b[:, 0] ** 2 + gravity_b[:, 1] ** 2


def body_collision(
    env: ManagerBasedRlEnv,
    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
) -> torch.Tensor:
    """Penalty for non-foot body contacts."""
    asset = env.scene[asset_cfg.name]
    return asset.data.body_contact_force_norm.sum(dim=1)


# --- Termination functions ---

def terminate_height_low(
    env: ManagerBasedRlEnv,
    min_height: float,
    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,
) -> torch.Tensor:
    asset = env.scene[asset_cfg.name]

def terminate_height_high(
    env: ManagerBasedRlEnv,

    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,

    asset = env.scene[asset_cfg.name]

def terminate_bad_orientation(
    env: ManagerBasedRlEnv,

    asset_cfg: SceneEntityCfg = _DEFAULT_ASSET_CFG,

    asset = env.scene[asset_cfg.name]
    gravity_b = asset.data.projected_gravity_b  # [B, 3]

def privileged_friction(

    task = get_task("Go2-Flat-v0")

    env_cfg.scene.num_envs = num_envs

    # Warmup

        wrapped.step(actions)

    # Benchmark
    torch.cuda.synchronize()
    t0 = time.perf_counter()
    for _ in range(num_steps):
        wrapped.step(actions)
    torch.cuda.synchronize()
    elapsed = time.perf_counter() - t0

    steps_per_sec = num_envs * num_steps / elapsed
    print(f"  num_envs={num_envs:>5d} | {num_steps} steps | "
          f"{elapsed:.2f}s | {steps_per_sec:,.0f} steps/sec")

# scripts/train.py
"""Train Go2 Teacher policy with PPO + asymmetric AC + DR/command curriculum."""
import sys
from pathlib import Path
sys.path.insert(0, str(Path(__file__).resolve().parent.parent.parent.parent))

import tyro
import torch
from dataclasses import dataclass

import mjlab.tasks  # noqa: F401
import applications.go2_mjlab.src.tasks  # noqa: F401

from mjlab.tasks.registry import get_task
from mjlab.rl import RslRlVecEnvWrapper

from applications.go2_mjlab.src.tasks.velocity.mdp.events import (
    DR_PHASE1_END,
    DR_PHASE2_END,
    make_dr_events_phase1,
    make_dr_events_phase2,
    make_dr_events_phase3,
)


@dataclass
class TrainArgs:
    task_id: str = "Go2-Flat-v0"
    num_envs: int = 1024
    max_iterations: int = 5000
    resume_path: str | None = None
    experiment_name: str = "go2_teacher"
    log_dir: str = "applications/go2_mjlab/results"


def get_dr_phase(iteration: int) -> int:
    if iteration < DR_PHASE1_END:
        return 1
    elif iteration < DR_PHASE2_END:
        return 2
    return 3


class DRCurriculumManager:
    """Manages DR phase transitions during training."""

    def __init__(self, env):
        self.env = env
        self.current_phase = 1

    def update(self, iteration: int):
        new_phase = get_dr_phase(iteration)
        if new_phase != self.current_phase:
            self.current_phase = new_phase
            if new_phase == 2:
                events = make_dr_events_phase2()
            else:
                events = make_dr_events_phase3()
            self.env.event_manager.update_events(events)
            print(f"  [DR] Switched to phase {new_phase} at iter {iteration}")


class CommandCurriculumManager:
    """Expands velocity command range when tracking is stable."""

    def __init__(self, env, initial_range=(0.3, 0.8), limit=(0.3, 1.5),
                 threshold=0.6, delta=0.1, patience=5):
        self.env = env
        self.range = list(initial_range)
        self.limit = limit
        self.threshold = threshold
        self.delta = delta
        self.patience = patience
        self.stable_count = 0

    def update(self, tracking_ratio: float):
        if tracking_ratio > self.threshold:
            self.stable_count += 1
        else:
            self.stable_count = max(0, self.stable_count - 1)

        if self.stable_count >= self.patience:
            self.stable_count = 0
            new_hi = min(self.limit[1], self.range[1] + self.delta)
            if new_hi != self.range[1]:
                self.range[1] = new_hi
                self.env.command_manager.update_ranges(
                    "twist", {"lin_vel_x": tuple(self.range)}
                )
                print(f"  [CMD] Expanded lin_vel_x to [{self.range[0]:.2f}, {self.range[1]:.2f}]")


def compute_tracking_ratio(env) -> float:
    """Compute velocity tracking ratio from latest step data."""
    asset = env.scene["robot"]
    vel_x = asset.data.root_lin_vel_b[:, 0]
    cmd_x = env.command_manager.get_command("twist")[:, 0]
    mask = cmd_x.abs() > 0.1
    if mask.sum() == 0:
        return 0.0
    ratio = (vel_x[mask] / cmd_x[mask]).clamp(0.0, 1.5)
    return ratio.mean().item()


def main():
    args = tyro.cli(TrainArgs)

    task = get_task(args.task_id)
    env_cfg = task.env_cfg
    env_cfg.scene.num_envs = args.num_envs
    rl_cfg = task.rl_cfg
    rl_cfg.max_iterations = args.max_iterations
    rl_cfg.experiment_name = args.experiment_name

    log_dir = Path(args.log_dir)
    log_dir.mkdir(parents=True, exist_ok=True)

    env = task.env_cls(env_cfg)
    wrapped = RslRlVecEnvWrapper(env)

    runner_cls = task.runner_cls
    runner = runner_cls(wrapped, rl_cfg, log_dir=str(log_dir), device="cuda:0")

    if args.resume_path:
        runner.load(args.resume_path)
        print(f"Resumed from {args.resume_path}")

    dr_mgr = DRCurriculumManager(env)
    cmd_mgr = CommandCurriculumManager(env)

    print(f"Training Go2 Teacher: {args.num_envs} envs, {args.max_iterations} iters")
    print(f"GPU: {torch.cuda.get_device_name(0)}")
    print(f"Log dir: {log_dir}")
    print()

    # Training loop with curriculum callbacks
    for iteration in range(args.max_iterations):
        # Update DR curriculum
        dr_mgr.update(iteration)

        # Run one PPO iteration (rollout + update)
        metrics = runner.train_step()

        # Update command curriculum
        tracking_ratio = compute_tracking_ratio(env)
        cmd_mgr.update(tracking_ratio)

        if (iteration + 1) % 50 == 0:
            avg_reward = metrics.get("mean_reward", 0.0)
            dr_phase = get_dr_phase(iteration)
            phase_name = ["", "no-DR", "light-DR", "full-DR"][dr_phase]
            print(
                f"Iter {iteration+1}/{args.max_iterations} | "
                f"reward: {avg_reward:.3f} | "
                f"track: {tracking_ratio:.2f} | "
                f"vx: [{cmd_mgr.range[0]:.2f},{cmd_mgr.range[1]:.2f}] | "
                f"{phase_name}"
            )

        if (iteration + 1) % rl_cfg.save_interval == 0:
            runner.save(str(log_dir / f"teacher_iter{iteration+1}.pt"))

    runner.save(str(log_dir / "teacher_final.pt"))
    env.close()
    print("Teacher training complete.")


if __name__ == "__main__":
    main()

    env_cfg.scene.num_envs = args.num_envs
    rl_cfg = task.rl_cfg

    env = task.env_cls(env_cfg)
    wrapped = RslRlVecEnvWrapper(env)

    runner = task.runner_cls(wrapped, rl_cfg, log_dir="/tmp/collect", device="cuda:0")
    runner.load(args.checkpoint)
    print(f"Loaded teacher: {args.checkpoint}")

    obs_dim = wrapped.observation_space.shape[0]
    history_length = args.history_length
    num_envs = args.num_envs

    # Rolling history buffer [num_envs, history_length, obs_dim]
    history_buf = torch.zeros(num_envs, history_length, obs_dim, device="cuda:0")

    all_histories = []
    all_actions = []
    collected = 0
    steps_needed = args.num_transitions // num_envs + 1

    obs, _ = wrapped.reset()

    print(f"Collecting {args.num_transitions} transitions ({steps_needed} steps x {num_envs} envs)...")

    for step in range(steps_needed):
        # Shift history and append current obs
        history_buf = torch.roll(history_buf, -1, dims=1)
        history_buf[:, -1, :] = obs

        # Get teacher action (deterministic)
        with torch.no_grad():
            actions = runner.get_inference_action(obs)

        # Store if history is full (after warmup)
        if step >= history_length:
            all_histories.append(history_buf.cpu().numpy())
            all_actions.append(actions.cpu().numpy())
            collected += num_envs

        # Step env
        obs, _, rewards, dones, extras = wrapped.step(actions)

        # Reset history for done envs
        done_ids = dones.nonzero(as_tuple=False).squeeze(-1)
        if done_ids.numel() > 0:
            history_buf[done_ids] = 0.0

        if (step + 1) % 100 == 0:
            print(f"  Step {step+1}/{steps_needed} | collected: {collected:,}")

        if collected >= args.num_transitions:
            break

    env.close()

    # Concatenate and save
    histories = np.concatenate(all_histories, axis=0)[:args.num_transitions]
    actions = np.concatenate(all_actions, axis=0)[:args.num_transitions]

    output_path = Path(args.output_path)
    output_path.parent.mkdir(parents=True, exist_ok=True)
    np.savez_compressed(str(output_path), obs_history=histories, actions=actions)
    print(f"Saved {histories.shape[0]} transitions to {output_path}")
    print(f"  obs_history shape: {histories.shape}")
    print(f"  actions shape: {actions.shape}")


if __name__ == "__main__":
    main()
```

# src/distill/student_network.py
"""Student policy: history encoder + action head."""
import torch
import torch.nn as nn


class StudentPolicy(nn.Module):
    """Student network: encodes obs history into latent, then predicts action.

    Architecture:
        obs_history (history_length, obs_dim) -> flatten -> MLP encoder -> latent (latent_dim)
        [latent, current_obs] -> MLP policy -> action (action_dim)
    """

    def __init__(
        self,
        obs_dim: int = 48,
        action_dim: int = 12,
        history_length: int = 20,
        latent_dim: int = 16,
        encoder_hidden: int = 128,
        policy_hidden: int = 128,
    ):
        super().__init__()
        self.obs_dim = obs_dim
        self.action_dim = action_dim
        self.history_length = history_length
        self.latent_dim = latent_dim

        # History encoder: (history_length * obs_dim) -> latent
        encoder_input_dim = history_length * obs_dim
        self.encoder = nn.Sequential(
            nn.Linear(encoder_input_dim, encoder_hidden),
            nn.ELU(),
            nn.Linear(encoder_hidden, encoder_hidden),
            nn.ELU(),
            nn.Linear(encoder_hidden, latent_dim),
        )

        # Policy head: (latent + current_obs) -> action
        policy_input_dim = latent_dim + obs_dim
        self.policy = nn.Sequential(
            nn.Linear(policy_input_dim, policy_hidden),
            nn.ELU(),
            nn.Linear(policy_hidden, policy_hidden),
            nn.ELU(),
            nn.Linear(policy_hidden, action_dim),
            nn.Tanh(),
        )

    def forward(self, obs_history: torch.Tensor) -> torch.Tensor:
        """Forward pass.

        Args:
            obs_history: [B, history_length, obs_dim] tensor

        Returns:
            actions: [B, action_dim] tensor in [-1, 1]
        """
        batch_size = obs_history.shape[0]
        current_obs = obs_history[:, -1, :]

        # Flatten history and encode
        flat_history = obs_history.reshape(batch_size, -1)
        latent = self.encoder(flat_history)

        # Policy head
        policy_input = torch.cat([latent, current_obs], dim=1)
        actions = self.policy(policy_input)
        return actions

    def get_latent(self, obs_history: torch.Tensor) -> torch.Tensor:
        """Get latent encoding only (for analysis)."""
        batch_size = obs_history.shape[0]
        flat_history = obs_history.reshape(batch_size, -1)
        return self.encoder(flat_history)
```

# src/distill/train_student.py
"""Offline supervised distillation: train student to mimic teacher actions."""
import sys
from pathlib import Path
sys.path.insert(0, str(Path(__file__).resolve().parent.parent.parent.parent.parent))

import tyro
import torch
import torch.nn as nn
import numpy as np
from dataclasses import dataclass
from torch.utils.data import DataLoader, TensorDataset

from applications.go2_mjlab.src.distill.student_network import StudentPolicy


@dataclass
class DistillArgs:
    dataset_path: str = "applications/go2_mjlab/results/distill_dataset.npz"
    output_path: str = "applications/go2_mjlab/results/student_final.pt"
    obs_dim: int = 48
    action_dim: int = 12
    history_length: int = 20
    latent_dim: int = 16
    lr: float = 1e-3
    batch_size: int = 256
    epochs: int = 200
    val_ratio: float = 0.1
    early_stop_patience: int = 15


def main():
    args = tyro.cli(DistillArgs)

    # Load dataset
    print(f"Loading dataset: {args.dataset_path}")
    data = np.load(args.dataset_path)
    obs_history = torch.FloatTensor(data["obs_history"])  # [N, H, obs_dim]
    actions = torch.FloatTensor(data["actions"])            # [N, action_dim]
    print(f"  Dataset: {obs_history.shape[0]} samples")

    # Train/val split
    n = obs_history.shape[0]
    n_val = int(n * args.val_ratio)
    n_train = n - n_val
    perm = torch.randperm(n)
    train_idx, val_idx = perm[:n_train], perm[n_train:]

    train_ds = TensorDataset(obs_history[train_idx], actions[train_idx])
    val_ds = TensorDataset(obs_history[val_idx], actions[val_idx])
    train_dl = DataLoader(train_ds, batch_size=args.batch_size, shuffle=True, pin_memory=True)
    val_dl = DataLoader(val_ds, batch_size=args.batch_size, shuffle=False, pin_memory=True)

    print(f"  Train: {n_train}, Val: {n_val}")

    # Create model
    device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
    model = StudentPolicy(
        obs_dim=args.obs_dim,
        action_dim=args.action_dim,
        history_length=args.history_length,
        latent_dim=args.latent_dim,
    ).to(device)
    print(f"  Model params: {sum(p.numel() for p in model.parameters()):,}")

    optimizer = torch.optim.Adam(model.parameters(), lr=args.lr)
    criterion = nn.MSELoss()

    # Training loop
    best_val_loss = float("inf")
    patience_counter = 0

    for epoch in range(args.epochs):
        # Train
        model.train()
        train_loss = 0.0
        for hist_batch, act_batch in train_dl:
            hist_batch = hist_batch.to(device)
            act_batch = act_batch.to(device)

            pred = model(hist_batch)
            loss = criterion(pred, act_batch)

            optimizer.zero_grad()
            loss.backward()
            optimizer.step()
            train_loss += loss.item() * hist_batch.shape[0]

        train_loss /= n_train

        # Validate
        model.eval()
        val_loss = 0.0
        with torch.no_grad():
            for hist_batch, act_batch in val_dl:
                hist_batch = hist_batch.to(device)
                act_batch = act_batch.to(device)
                pred = model(hist_batch)
                loss = criterion(pred, act_batch)
                val_loss += loss.item() * hist_batch.shape[0]
        val_loss /= n_val

        if (epoch + 1) % 10 == 0:
            print(f"  Epoch {epoch+1}/{args.epochs} | "
                  f"train_loss: {train_loss:.6f} | val_loss: {val_loss:.6f}")

        # Early stopping
        if val_loss < best_val_loss:
            best_val_loss = val_loss
            patience_counter = 0
            torch.save(model.state_dict(), args.output_path)
        else:
            patience_counter += 1
            if patience_counter >= args.early_stop_patience:
                print(f"  Early stopping at epoch {epoch+1} (patience={args.early_stop_patience})")
                break

    print(f"\nBest val loss: {best_val_loss:.6f}")
    print(f"Saved to: {args.output_path}")


if __name__ == "__main__":
    main()
```

sys.path.insert(0, str(Path(__file__).resolve().parent.parent.parent.parent.parent))

import numpy as np

    action_dim: int = 12
    history_length: int = 20
    latent_dim: int = 16
    opset_version: int = 17

        action_dim=args.action_dim,
        history_length=args.history_length,
        latent_dim=args.latent_dim,

        output_names=["action"],

            "action": {0: "batch_size"},

    with torch.no_grad():
        pt_output = model(torch.FloatTensor(test_input)).numpy()

    max_diff = np.abs(onnx_output - pt_output).max()
    print(f"ONNX vs PyTorch max diff: {max_diff:.8f}")
    assert max_diff < 1e-5, f"ONNX verification failed: max_diff={max_diff}"
    print("ONNX verification PASSED")


if __name__ == "__main__":
    main()
